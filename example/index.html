react-colorpickr

A colorpicker for React that's just a colorpicker. Fork it on GitHub.

Install

npm install react-colorpickr

You'll also want to include a copy of react-colorpickr.css in your code.

Usage

var React = require('react');
var ColorPicker = require('react-colorpickr');

var App = React.createClass({
  onChange: function(color) {
    console.log(color);
  },

  render: function() {
    return (
      <div>
        <ColorPicker onChange={this.onChange} />
      </div>
    );
  }
});
